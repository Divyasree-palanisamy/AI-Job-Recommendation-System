# Email Configuration for Password Reset
# IMPORTANT: Rename this file to '.env' for it to work!

# Step 1: Enable 2-Factor Authentication on Gmail
# Go to: https://myaccount.google.com/security
# Enable 2-Step Verification

# Step 2: Generate App Password
# Go to: https://myaccount.google.com/apppasswords
# Select "Mail" and "Other (custom name)"
# Enter "Career Portal" as the name
# Copy the 16-character password

# Step 3: Replace the values below with your real credentials
MAIL_SERVER=smtp.gmail.com
MAIL_PORT=587
MAIL_USE_TLS=True
MAIL_USE_SSL=False
MAIL_USERNAME=your_actual_gmail@gmail.com
MAIL_PASSWORD=your_16_character_app_password
MAIL_DEFAULT_SENDER=noreply@careerguidance.com
